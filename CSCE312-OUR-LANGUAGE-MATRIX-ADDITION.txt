Main:
	irmov $0x4, %rA
	rrmov %rA, %rE
	irmov array1 location, %rB
	irmov array2 location, %rC
	irmov array3 location, %rD
	and %rA %rA
Adder:
	ld (%rB), %rE
	ld (%rC), %rF
	add %rE, %rF
	st %rE, (%rD)
	add %rE, %rB
	add %rE, %rC
	add %rE, %rD
	irmov, $0x-1, %rE
	add %rE, %rA
	jne Adder
	
.pos 0x100
